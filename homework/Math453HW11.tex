\question \textbf{M.34}
Let $f(x)=[1]x\in\mathbb{Z}_6[x]$. Find a polynomial $p(x)$ in $\mathbb{Z}_6[x]$ such that $p(x)\ne f(x)$, but $p\colon \mathbb{Z}_6\to \mathbb{Z}_6$ defines the same function as $f\colon \mathbb{Z}_6\to \mathbb{Z}_6$.


\question \textbf{CORE M.35}
\begin{parts}
\part Find polynomials $p(x),q(x)\in\mathbb{Z}_4[x]$ of degree 2 such that $p(x)q(x)$ has degree $4$. 
\part Find polynomials $p(x),q(x)\in\mathbb{Z}_4[x]$ of degree 2 such that $p(x)q(x)$ has degree $3$.
\part Find polynomials $p(x),q(x)\in\mathbb{Z}_4[x]$ of degree 2 such that $p(x)q(x)$ has degree $2$.
\part Find polynomials $p(x),q(x)\in\mathbb{Z}_4[x]$ of degree 2 such that $p(x)q(x)$ has degree $1$.
\part Find polynomials $p(x),q(x)\in\mathbb{Z}_4[x]$ of degree 2 such that $p(x)q(x)$ has degree $0$.
\end{parts}
